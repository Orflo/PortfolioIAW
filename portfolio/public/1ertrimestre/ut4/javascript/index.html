            var num1 = parseInt(prompt ("numero 1"));
            var num2 = parseInt(prompt ("numero 2"));
            //metodo sumar
            function sumar () {
                return num1+num2;
            }
            //metodo resta
            function resta () {
                return num1-num2;
            }
            //metodo mult
            function mult() {
                return num1*num2;
            }
            //metodo div
            function div () {
                return num1/num2;
            }
        </script>
    </head>
    <body>
        <input type="button" onclick="alert(sumar(num1,num2))" value="sumar">
        <input type="button" onclick="alert(resta(num1,num2))" value="restar">
        <input type="button" onclick="alert(div(num1,num2))" value="dividir">
        <input type="button" onclick="alert(mult(num1,num2))" value="multiplicar">
    </body>

                    if(num1 > num2){
                        return num1+" es mayor que "+num2;
                    }else if(num1 < num2){
                        return num1+" es menor que "+num2;
                    }else{
                        return num1+" es igual a "+num2;
                    }
                }

            function listar (){
                lista = [1, 2, 13, 4, 25];
                num1 = 0
                for(var i=0; i< lista.length; i++){
                    num2 = lista[i]
                    if(num2 > num1){
                        num1 = num2
                    }
                }
                alert(num1)
            }
            listar();
